<script src="https://cdnjs.cloudflare.com/ajax/libs/PapaParse/5.4.1/papaparse.min.js"></script>
<script>
const search = document.getElementById('search');
const country = document.getElementById('country');
const level = document.getElementById('level');

function filter(){
  const q = search.value.trim().toLowerCase();
  const c = country.value;
  const l = level.value;
  const cards = [...document.querySelectorAll('.card')];
  cards.forEach(card=>{
    const text = card.textContent.toLowerCase();
    const matchQ = !q || text.includes(q);
    const matchC = !c || card.dataset.country === c;
    const matchL = !l || card.dataset.level === l;
    card.style.display = (matchQ && matchC && matchL)?'':'none';
  });
}

search.addEventListener('input',filter);
country.addEventListener('change',filter);
level.addEventListener('change',filter);

// Google Sheet
const sheetUrl = 'YOUR_CSV_GOOGLE_SHEET_URL';

Papa.parse(sheetUrl,{
  download:true,
  header:true,
  complete:function(results){
    const list = document.getElementById('list');
    list.innerHTML='';
    results.data.forEach(s=>{
      if(!s.Name) return;
      const card = document.createElement('article');
      card.className='card';
      card.dataset.country=s.Country;
      card.dataset.level=s.Level;
      card.innerHTML=`
        <h3>${s.Name}</h3>
        <div class="meta">${s.Country} · ${s.Level} · ${s.Funding} · Deadline: ${s.Deadline}</div>
        <p>${s.Notes}</p>
        <a class="cta" href="${s.Link}" target="_blank" rel="noopener">Visit official page</a>
      `;
      list.appendChild(card);
    });
  }
});
</script>
